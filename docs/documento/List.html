def extraer_head       
        valor = @head.value 
        if(@head==@tail)
                @head,@tail=nil
        else  
                @head = @head[:next]
                @head[:prev] = nil
        end
        return valor
end

def to_s
        cadena = ""
        cabeza = @head
        until (cabeza == nil) do
                cadena += "#{cabeza.value.to_s}"
                cabeza = cabeza.next
        end
        return cadena
end
